
function(add_host_tool _tool)
    add_executable(${_tool} ${ARGN})
    if(NEW_STYLE_BUILD)
        set_target_properties(${_tool} PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${TOOLS_FOLDER})
    endif()
endfunction()
